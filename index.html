Demonstration on the use of Pager class

The class is contained in script.js

Get this file from the hiteshlala/pager repo

Embed in your html:

<!DOCTYPE html>
<html lang="en">
  <head></head>
  <body>
    <div id="pager"></div>
    <script src="pager.js"></script>
  </body>
</html>

Instantiate the pager:

<script>
  let pager;
  const numberOfPages = 52;
  const pagesDisplayWindow = 6;
  function updatePageCallback(page) {
    console.log('Load something for page number', 
      page);
  }
  window.onload = function() {
    pager = new Pager(
      numberOfPages, 
      pagesDisplayWindow, 
      updatePageCallback, 
      'pager');
  };
</script>

To adjust the styles add a JSON with style overides. As shown below:

let pager;
const numberOfPages = 52;
const pagesDisplayWindow = 4;
function updatePageCallback(page) {
  console.log('Load something for page number', 
    page);
}          
const overideStyles = { 
  pager: '{ justify-content: center }',
  'pager-button': `{ border: 1px solid #6c9393; 
        border-radius: 5px; background-color: white;}`,
  clickable: '{ background-color: #d0d0d0; }',
  'clickable:hover': '{border: 1px solid #f2d2f3;}',
  selected: '{ color: #5353c6 }'
};
window.onload = function() {
  pager = new Pager(
    numberOfPages, 
    pagesDisplayWindow, 
    updatePageCallback, 
    'pager', 
    overideStyles);
};
